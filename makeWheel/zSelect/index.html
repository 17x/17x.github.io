example 2


if you want set multiple default selected 
you should add a multiple attribute for select and add a selected attribute for each option
<select name="gender" id=""  zSelect='{
     "triggerOpt":"click",
     "multiple":true,
     "filter":true,
     "visibility":"visibility",
     "top":10,
     "left":400
 }' multiple>
    <option value="b00001">b000001</option>
    if you want set multiple default selected
    <option value="b00002" selected>b000002</option>
    <option value="b00003" selected>b000003</option>
    <option value="b00004" selected>b000004</option>
