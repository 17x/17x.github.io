### Bubble 冒泡排序
***

给定数组 

```
    let arr = [7, 9, 6, 4, 5, 8, 1, 3, 2];
    
    //外层遍历
    for (let i = 0; i < arr.length; i++) {   
        //从最右侧开始创建一个游标
        //往左移动
        for (let j = arr.length - 1; j >= 0; j--) {
            //判断左侧
            if(isNaN(arr[j - 1]))continue;
            //右边和左边进行对比
            //如果小于左边 则 对换位置
            if (arr[j] < arr[j - 1]) {
                let right = arr[j];
				arr[j] = arr[j - 1];
				arr[j - 1] = right;
            }
        }
    }
    
// 一次遍历
     [7, 9, 6, 4, 5, 8, 1, 3, 2]
                           |__|
     [7, 9, 6, 4, 5, 8, 1, 2, 3]
                        |__|                        
     [7, 9, 6, 4, 5, 8, 1, 2, 3]
                     |__|
     [7, 9, 6, 4, 5, 1, 8, 2, 3]
                  |__|
     [7, 9, 6, 4, 1, 5, 8, 2, 3]
               |__|
     [7, 9, 6, 1, 4, 5, 8, 2, 3]

     [7, 9, 1, 6, 4, 5, 8, 2, 3]
```
